

Struts + Freemarker

	1、XxAction.java
	
		@Action(value = "student3001", results = { @Result(type="freemarker", name = SUCCESS, location = "/business/dd/studentlist.ftl") })
		public String tolist() {
			classesList = ddService.findByType(Dd.TYPE_BJ);
			return SUCCESS;
		}
		
	2、WebContent/business/dd/studentlist.ftl
	
		<#list classesList as cla>
			${cla.id?c} -- ${cla.name?html}
		</#list>
	
	
Struts-Tags + Freemarker
	
	1、添加 WEB-INF/struts-tags.tld

	2、WEB-INF/web.xml
	
		<servlet>
			<servlet-name>JspSupportServlet</servlet-name>
			<servlet-class>org.apache.struts2.views.JspSupportServlet</servlet-class>
			<load-on-startup>1</load-on-startup>
		</servlet>
		
	3、WebContent/business/dd/studentlist.ftl
	
		<#assign s=JspTaglibs["/WEB-INF/struts-tags.tld"] />
		<@s.property />
